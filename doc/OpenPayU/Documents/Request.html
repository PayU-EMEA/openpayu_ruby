def set_headers
  @headers = {
    'OpenPayu-Signature' => generate_signature_structure(
        @body, 
        OpenPayU::Configuration.algorithm, 
        OpenPayU::Configuration.merchant_pos_id, 
        OpenPayU::Configuration.signature_key
      ),
    'openpayu-signature' => generate_signature_structure(
        @body, 
        OpenPayU::Configuration.algorithm, 
        OpenPayU::Configuration.merchant_pos_id, 
        OpenPayU::Configuration.signature_key
      )
  }
end
